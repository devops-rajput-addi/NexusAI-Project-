# DevOps MCP Hub - Environment Configuration

# Copy this file to .env and fill in your values

# -------------------------------------------
# JIRA CONFIGURATION [REQUIRED]
# -------------------------------------------

# Your Jira instance URL
# For Jira Cloud: https://your-company.atlassian.net
# For Jira Server: https://jira.your-company.com
JIRA_BASE_URL=https://jira.autodesk.com

# Your email address (required for Jira Cloud only)
# Leave empty for Jira Server with PAT authentication
JIRA_EMAIL=

# API Token or Personal Access Token
# Jira Cloud: Generate at https://id.atlassian.com/manage-profile/security/api-tokens
# Jira Server: Generate in Jira > Profile > Personal Access Tokens
JIRA_API_TOKEN=your_jira_pat_token

# -------------------------------------------
# GITHUB CONFIGURATION [OPTIONAL]
# -------------------------------------------
# Only token is needed - no username or email required!

# GitHub Personal Access Token
# Generate at: https://github.com/settings/tokens
# Required scopes: repo, read:user
GITHUB_TOKEN=your_github_token

# -------------------------------------------
# CONFLUENCE CLOUD CONFIGURATION [OPTIONAL]
# -------------------------------------------
# NOTE: If API tokens are blocked by admin, use the 
# Documentation Tools below instead (save to GitHub/local files)

# Your Confluence Cloud URL
CONFLUENCE_BASE_URL=https://your-company.atlassian.net

# Your Atlassian account email (required for Confluence Cloud)
CONFLUENCE_EMAIL=your.email@company.com

# API Token (may be blocked by admin)
CONFLUENCE_API_TOKEN=your_atlassian_api_token

# Restrict to a specific space (recommended)
CONFLUENCE_SPACE_KEY=YOUR_SPACE_KEY

# -------------------------------------------
# DOCUMENTATION OUTPUT [ALTERNATIVE TO CONFLUENCE]
# -------------------------------------------
# If Confluence API is blocked, documentation can be 
# saved to local files or GitHub repositories

# Local directory for generated documentation
DOCS_OUTPUT_DIR=./generated-docs

# Default GitHub repo for documentation (optional)
# Used by save_*_to_github tools
DOCS_GITHUB_OWNER=your-github-org
DOCS_GITHUB_REPO=your-docs-repo
